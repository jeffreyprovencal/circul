<script>
let adminToken = localStorage.getItem('circul_admin_token');
let adminData = null;
let currentTab = 'overview';

// Auto-login if token exists
window.onload = () => {
  if (adminToken) {
    showMain();
    loadCurrentTab();
  }
};

// Enter key on login
document.getElementById('login-password').addEventListener('keydown', e => { if (e.key === 'Enter') adminLogin(); });

async function adminLogin() {
  const email = document.getElementById('login-email').value.trim();
  const password = document.getElementById('login-password').value;
  const errEl = document.getElementById('login-error');
  errEl.textContent = '';
  if (!email || !password) { errEl.textContent = 'Email and password required'; return; }
  try {
    const res = await fetch('/api/admin/login', {
      method: 'POST', headers: {'Content-Type':'application/json'},
      body: JSON.stringify({ email, password })
    });
    const data = await res.json();
    if (!data.success) { errEl.textContent = data.message || 'Login failed'; return; }
    adminToken = data.token;
    adminData = data.admin;
    localStorage.setItem('circul_admin_token', adminToken);
    showMain();
    loadCurrentTab();
  } catch (e) {
    errEl.textContent = 'Network error. Try again.';
  }
}

function showMain() {
  document.getElementById('login-screen').style.display = 'none';
  document.getElementById('main-screen').style.display = 'flex';
  const stored = localStorage.getItem('circul_admin_token');
  if (stored && !adminData) {
    // try to decode email from token
    try {
      const b64 = stored.split('.')[0];
      adminData = JSON.parse(atob(b64.replace(/-/g,'+').replace(/_/g,'/')));
    } catch {}
  }
  if (adminData) document.getElementById('admin-email-display').textContent = adminData.email || 'Admin';
}

function logout() {
  localStorage.removeItem('circul_admin_token');
  adminToken = null;
  document.getElementById('main-screen').style.display = 'none';
  document.getElementById('login-screen').style.display = 'flex';
}

function authHeaders() {
  return { 'Content-Type': 'application/json', 'Authorization': 'Bearer ' + adminToken };
}

async function apiFetch(url, opts = {}) {
  const res = await fetch(url, { headers: authHeaders(), ...opts });
  if (res.status === 401) { logout(); return null; }
  return res.json();
}

function showTab(tab, el) {
  document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
  el.classList.add('active');
  document.querySelectorAll('[id^=tab-]').forEach(t => t.style.display = 'none');
  document.getElementById('tab-' + tab).style.display = '';
  currentTab = tab;
  loadTab(tab);
}

function loadCurrentTab() { loadTab(currentTab); }

function loadTab(tab) {
  if (tab === 'overview') loadOverview();
  else if (tab === 'collectors') loadCollectors();
  else if (tab === 'transactions') loadTransactions();
  else if (tab === 'operators') loadOperators();
  else if (tab === 'buyers') loadBuyers();
  else if (tab === 'reputation') loadReputation();
}

async function loadOverview() {
  const data = await apiFetch('/api/admin/stats');
  if (!data || !data.success) return;
  const s = data.stats;

  // Role breakdown from operators list
  const opsData = await apiFetch('/api/admin/operators');
  const roleCounts = { collector:0, aggregator:0, processor:0, converter:0, operator:0, admin:0 };
  if (opsData && opsData.operators) {
    opsData.operators.forEach(o => { if (roleCounts[o.role] !== undefined) roleCounts[o.role]++; });
  }

  document.getElementById('stat-grid').innerHTML = `
    <div class="stat-card"><div class="num">${s.collectors}</div><div class="label">Collectors (USSD)</div></div>
    <div class="stat-card"><div class="num">${roleCounts.aggregator}</div><div class="label">Aggregators</div></div>
    <div class="stat-card"><div class="num">${roleCounts.processor}</div><div class="label">Processors</div></div>
    <div class="stat-card"><div class="num">${roleCounts.converter}</div><div class="label">Converters</div></div>
    <div class="stat-card"><div class="num">${s.transactions}</div><div class="label">Transactions</div></div>
    <div class="stat-card"><div class="num">${parseFloat(s.total_volume_kg).toFixed(1)}</div><div class="label">Total kg collected</div></div>
  `;
  const materials = ['PET','HDPE','LDPE','PP'];
  const matMap = {};
  (s.by_material || []).forEach(m => matMap[m.material_type] = m);
  document.getElementById('mat-grid').innerHTML = materials.map(m => {
    const d = matMap[m] || {};
    return `<div class="mat-card">
      <div class="mat-name">${m}</div>
      <div class="mat-kg">${parseFloat(d.total_kg || 0).toFixed(1)} kg</div>
      <div class="mat-txns">${d.count || 0} transactions</div>
    </div>`;
  }).join('');

  // System health
  const txnData = await apiFetch('/api/admin/transactions?limit=1');
  const paySuccess = s.transactions > 0 ? 'N/A' : '—';
  document.getElementById('system-health-content').innerHTML = `
    <div style="display:grid;grid-template-columns:repeat(3,1fr);gap:16px">
      <div><div style="font-size:1.5rem;font-weight:700;color:var(--green-dark)">${s.transactions}</div><div style="font-size:0.82rem;color:var(--slate-light)">Total DB transactions</div></div>
      <div><div style="font-size:1.5rem;font-weight:700;color:var(--green-dark)">${s.operators}</div><div style="font-size:0.82rem;color:var(--slate-light)">Active operators (all roles)</div></div>
      <div><div style="font-size:1.5rem;font-weight:700;color:var(--green-dark)">${s.buyers}</div><div style="font-size:0.82rem;color:var(--slate-light)">Legacy buyer accounts</div></div>
    </div>
  `;
}

async function loadCollectors() {
  const data = await apiFetch('/api/admin/collectors');
  if (!data || !data.success) return;
  const tbody = document.getElementById('collectors-tbody');
  if (!data.collectors.length) { tbody.innerHTML = '<tr><td colspan="7" class="empty">No collectors yet.</td></tr>'; return; }
  tbody.innerHTML = data.collectors.map(c => `
    <tr>
      <td>${escHtml(c.first_name + ' ' + c.last_name)}</td>
      <td>${escHtml(c.phone || '—')}</td>
      <td>${escHtml(c.region || '—')}</td>
      <td>${c.average_rating ? '⭐ ' + parseFloat(c.average_rating).toFixed(1) : '—'}</td>
      <td>${parseFloat(c.total_weight_kg).toFixed(2)} kg</td>
      <td>${c.transaction_count}</td>
      <td>${fmtDate(c.created_at)}</td>
    </tr>`).join('');
}

async function loadReputation() {
  const data = await apiFetch('/api/admin/operators');
  if (!data || !data.success) return;
  const tbody = document.getElementById('reputation-tbody');
  const ops = data.operators.filter(o => ['aggregator','processor','converter','collector'].includes(o.role));
  if (!ops.length) { tbody.innerHTML = '<tr><td colspan="6" class="empty">No operators with reputation data.</td></tr>'; return; }

  // Fetch ratings for each operator in parallel (cap at 20)
  const slice = ops.slice(0, 20);
  const ratingsData = await Promise.all(slice.map(op =>
    fetch(`/api/ratings/operator/${op.id}`, { headers: authHeaders() }).then(r => r.json()).catch(() => ({ success: false }))
  ));

  tbody.innerHTML = slice.map((op, i) => {
    const rd = ratingsData[i];
    const avg = rd && rd.summary ? rd.summary.avg_rating || '—' : '—';
    const cnt = rd && rd.summary ? rd.summary.count || 0 : 0;
    return `<tr>
      <td>${escHtml(op.name)}</td>
      <td><span class="badge ${ROLE_BADGE_CLASS[op.role] || 'badge-gray'}">${op.role}</span></td>
      <td>${escHtml(op.city || '—')}</td>
      <td>${avg !== '—' ? '⭐ ' + avg : '—'}</td>
      <td>${cnt}</td>
      <td>
        <button class="btn btn-danger btn-sm" onclick="flagOperator(${op.id})" title="Flag for review">Flag</button>
      </td>
    </tr>`;
  }).join('');
}

async function flagOperator(id) {
  if (!confirm('Flag this operator for review? This will mark them as flagged but keep them active.')) return;
  await apiFetch(`/api/admin/operators/${id}`, { method: 'PUT', body: JSON.stringify({ is_flagged: true }) });
  loadReputation();
}

async function loadBuyers() {

  if (!data || !data.success) return;
  const tbody = document.getElementById('buyers-tbody');
  if (!data.buyers.length) { tbody.innerHTML = '<tr><td colspan="7" class="empty">No buyers yet. Add aggregators and processors above.</td></tr>'; return; }

    const priceCount = (b.prices || []).length;

  }).join('');
}

<script>
var state = { operator: null };

var ROLE_ROUTES = {
    collector: '/collector-dashboard',
    aggregator: '/aggregator-dashboard',
    processor: '/processor-dashboard',
    converter: '/converter-dashboard'
};

var MAT_COLORS = { PET: '#16a34a', HDPE: '#0891b2', LDPE: '#7c3aed', PP: '#f59e0b' };

function checkAuth() {
    var saved = localStorage.getItem('circul_operator');
    if (saved) {
        try {
            state.operator = JSON.parse(saved);
            if (state.operator.role !== 'collector') {
                var route = ROLE_ROUTES[state.operator.role];
                if (route) { window.location.href = route; return; }
            }
            showDashboard();
            return;
        } catch (e) { localStorage.removeItem('circul_operator'); }
    }
    document.getElementById('loginOverlay').classList.remove('hidden');
}

function showDashboard() {
    document.getElementById('loginOverlay').classList.add('hidden');
    document.getElementById('dashboard').style.display = '';
    var op = state.operator;
    document.getElementById('navName').textContent = op.name + (op.company ? ' · ' + op.company : '');
    // Report link — try to use operator ID for now; collector ID mapping not available at login
    document.getElementById('reportLink').href = '/report?operator_id=' + op.id;
    loadStats();
}

function handleLogin(e) {
    e.preventDefault();
    var phone = document.getElementById('loginPhone').value.trim();
    var pin = document.getElementById('loginPin').value.trim();
    if (!phone || !pin) return;
    var btn = document.getElementById('loginBtn');
    btn.disabled = true; btn.innerHTML = '<div class="spinner"></div>';
    document.getElementById('loginError').classList.remove('visible');
    fetch('/api/operators/login', {
        method: 'POST', headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ phone: phone, pin: pin })
    })
    .then(function(r) { return r.json(); })
    .then(function(data) {
        if (data.success) {
            state.operator = data.operator;
            localStorage.setItem('circul_operator', JSON.stringify(data.operator));
            if (data.operator.role !== 'collector') {
                var route = ROLE_ROUTES[data.operator.role];
                if (route) { window.location.href = route; return; }
            }
            showDashboard();
        } else {
            var e = document.getElementById('loginError');
            e.textContent = data.message || 'Invalid phone or PIN';
            e.classList.add('visible');
        }
    })
    .catch(function() {
        var e = document.getElementById('loginError');
        e.textContent = 'Connection error. Please try again.';
        e.classList.add('visible');
    })
    .finally(function() { btn.disabled = false; btn.textContent = 'Log In'; });
}

function handleLogout() {
    state.operator = null;
    localStorage.removeItem('circul_operator');
    window.location.href = '/dashboard';
}

function loadStats() {
    var id = state.operator.id;
    fetch('/api/operators/' + id + '/dashboard-stats')
        .then(function(r) { return r.json(); })
        .then(function(data) {
            if (!data.success) return;
            var s = data.stats;
            document.getElementById('statTotalKg').textContent = fmt(s.totals.total_kg);
            document.getElementById('statEarned').textContent = fmt(s.totals.total_value);
            document.getElementById('statTxns').textContent = s.totals.total_txns;
            document.getElementById('statMonthKg').textContent = fmt(s.this_month.month_kg);
            renderMaterials(s.by_material);
            renderRecentTxns(s.payment_history);
        });
    loadPrices();
}

function renderMaterials(mats) {
    var c = document.getElementById('materialBreakdown');
    if (!mats || !mats.length) { c.innerHTML = '<div class="empty">No transactions yet</div>'; return; }
    var total = mats.reduce(function(s, m) { return s + parseFloat(m.kg); }, 0);
    var max = parseFloat(mats[0].kg);
    var html = '';
    mats.forEach(function(m) {
        var kg = parseFloat(m.kg);
        var pct = total > 0 ? (kg / total * 100) : 0;
        var bw = max > 0 ? (kg / max * 100) : 0;
        var color = MAT_COLORS[m.material_type] || '#6b7280';
        html += '<div class="mat-row"><span class="mat-label">' + m.material_type + '</span><div class="mat-bar-track"><div class="mat-bar-fill" style="width:' + bw.toFixed(1) + '%;background:' + color + '"></div></div><span class="mat-kg">' + fmt(kg) + ' kg</span><span class="mat-pct">' + pct.toFixed(1) + '%</span></div>';
    });
    c.innerHTML = html;
}

function renderRecentTxns(history) {
    var tbody = document.getElementById('txnBody');
    if (!history || !history.length) { tbody.innerHTML = '<tr><td colspan="6" class="empty">No transactions yet</td></tr>'; return; }
    tbody.innerHTML = history.map(function(p) {
        var statusClass = p.status === 'success' ? 'badge-green' : p.status === 'processing' ? 'badge-amber' : 'badge-gray';
        return '<tr><td>' + fmtDate(p.created_at) + '</td><td>—</td><td>—</td><td>GHS ' + fmt(p.amount) + '</td><td><span class="badge ' + statusClass + '">' + (p.status || 'unpaid') + '</span></td><td>—</td></tr>';
    }).join('');
    // If no payment history, try fetching transactions directly
    if (!history.length) fetchRecentTxns();
}

function fetchRecentTxns() {
    var id = state.operator.id;
    fetch('/api/transactions?operator_id=' + id + '&limit=20')
        .then(function(r) { return r.json(); })
        .then(function(data) {
            if (!data.success || !data.transactions.length) return;
            var tbody = document.getElementById('txnBody');
            tbody.innerHTML = data.transactions.map(function(t) {
                var statusClass = t.payment_status === 'paid' ? 'badge-green' : t.payment_status === 'pending' ? 'badge-amber' : 'badge-gray';
                var collector = (t.collector_first_name || '') + ' ' + (t.collector_last_name || '');
                return '<tr><td>' + fmtDate(t.transaction_date) + '</td><td><span class="badge badge-green">' + t.material_type + '</span></td><td>' + fmt(t.net_weight_kg) + '</td><td>' + (t.total_price ? 'GHS ' + fmt(t.total_price) : '—') + '</td><td><span class="badge ' + statusClass + '">' + (t.payment_status || 'unpaid') + '</span></td><td>—</td></tr>';
            }).join('');
        });
}

function loadPrices() {
    var city = state.operator.city || 'Accra';
    document.getElementById('priceCity').textContent = city;
    fetch('/api/prices?role=collector&city=' + encodeURIComponent(city))
        .then(function(r) { return r.json(); })
        .then(function(data) {
            var panel = document.getElementById('pricePanel');
            if (!data.success) { panel.innerHTML = '<div class="empty">No prices available</div>'; return; }
            var near = data.near_prices || [];
            var national = data.national_averages || [];
            var html = '';
            if (near.length) {
                html += '<p style="font-size:0.78rem;font-weight:600;color:var(--green-primary);margin-bottom:12px;text-transform:uppercase;letter-spacing:0.5px;">Near you (' + city + ')</p>';
                html += '<table style="width:100%;font-size:0.85rem;border-collapse:collapse;margin-bottom:16px;">';
                html += '<thead><tr><th style="text-align:left;padding:6px 0;color:var(--slate-light);font-size:0.75rem;font-weight:600;border-bottom:1px solid var(--border)">Material</th><th style="text-align:right;padding:6px 0;color:var(--slate-light);font-size:0.75rem;font-weight:600;border-bottom:1px solid var(--border)">USD/kg</th><th style="text-align:right;padding:6px 0;color:var(--slate-light);font-size:0.75rem;font-weight:600;border-bottom:1px solid var(--border)">GHS/kg</th><th style="text-align:right;padding:6px 0;color:var(--slate-light);font-size:0.75rem;font-weight:600;border-bottom:1px solid var(--border)">Actual avg</th></tr></thead><tbody>';
                near.forEach(function(p) {
                    html += '<tr><td style="padding:8px 0;font-weight:600;color:var(--green-deep)">' + p.material_type + '</td><td style="padding:8px 0;text-align:right">$' + parseFloat(p.price_per_kg_usd).toFixed(4) + '</td><td style="padding:8px 0;text-align:right;color:var(--green-primary);font-weight:600">' + (p.price_per_kg_ghs ? 'GHS ' + parseFloat(p.price_per_kg_ghs).toFixed(2) : '—') + '</td><td style="padding:8px 0;text-align:right;color:var(--slate-light)">' + (p.actual_avg_price_30d ? 'GHS ' + parseFloat(p.actual_avg_price_30d).toFixed(2) : '—') + '</td></tr>';
                });
                html += '</tbody></table>';
            }
            if (national.length) {
                html += '<p style="font-size:0.78rem;font-weight:600;color:var(--slate-light);margin-bottom:8px;text-transform:uppercase;letter-spacing:0.5px;">National average</p>';
                national.forEach(function(n) {
                    html += '<div style="display:flex;justify-content:space-between;padding:6px 0;font-size:0.85rem;border-bottom:1px solid var(--border)"><span style="font-weight:600">' + n.material_type + '</span><span style="color:var(--slate-light)">$' + parseFloat(n.avg_usd).toFixed(4) + '/kg · ' + n.buyer_count + ' buyers</span></div>';
                });
            }
            if (!near.length && !national.length) {
                html = '<div class="empty">No prices posted yet for your area</div>';
            }
            panel.innerHTML = html;
        })
        .catch(function() {
            document.getElementById('pricePanel').innerHTML = '<div class="empty">Unable to load prices</div>';
        });
}

function fmt(n) {
    var v = parseFloat(n);
    if (isNaN(v)) return '0';
    if (v >= 1000) return v.toLocaleString('en-US', { maximumFractionDigits: 1 });
    return v.toFixed(2);
}

function fmtDate(iso) {
    if (!iso) return '—';
    var d = new Date(iso);
    return d.toLocaleDateString('en-GB', { day: '2-digit', month: 'short', year: 'numeric' });
}

function checkAuth() {

function handleLogin(e) {

            '<button class="price-post-btn" onclick="postPrice(\'' + mat + '\')">Post</button>' +

function postPrice(mat) {

    if (!usd || parseFloat(usd) <= 0) { alert('Enter a valid USD price'); return; }
    var ghs = (rate && parseFloat(rate) > 0) ? (parseFloat(usd) * parseFloat(rate)).toFixed(2) : null;
    fetch('/api/prices', {
        method: 'POST', headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ operator_id: state.operator.id, material_type: mat, price_per_kg_usd: parseFloat(usd), price_per_kg_ghs: ghs ? parseFloat(ghs) : null, usd_to_ghs_rate: rate ? parseFloat(rate) : null })

function setStar(n) {

        s.classList.toggle('lit', parseInt(s.getAttribute('data-r')) <= n);
    });

function toggleTag(chip) {
    var tag = chip.getAttribute('data-tag');
    var type = chip.getAttribute('data-type');
    var idx = state.selectedTags.indexOf(tag);
    if (idx >= 0) { state.selectedTags.splice(idx, 1); chip.classList.remove('selected-pos', 'selected-neg'); }
    else { state.selectedTags.push(tag); chip.classList.add(type === 'pos' ? 'selected-pos' : 'selected-neg'); }
}

function submitRating() {
    if (!state.currentRatingStar) return;
    fetch('/api/ratings/operator', {
        method: 'POST', headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            rater_operator_id: state.operator.id,
            rated_collector_id: state.currentRatingCollectorId,
            rating: state.currentRatingStar,
            tags: state.selectedTags,
            notes: document.getElementById('ratingNote').value || null,
            rating_direction: 'aggregator_rates_collector'
        })
    })
    .then(function(r) { return r.json(); })
    .then(function(data) { if (data.success) { closeRatingModal(); } else { alert(data.message); } });
}

function fmt(n) {
    var v = parseFloat(n);
    if (isNaN(v)) return '0';
    if (v >= 10000) return v.toLocaleString('en-US', { maximumFractionDigits: 0 });
    if (v >= 100) return v.toLocaleString('en-US', { maximumFractionDigits: 1 });
    return v.toFixed(2);
}

function fmtDate(iso) {
    if (!iso) return '—';
    return new Date(iso).toLocaleDateString('en-GB', { day: '2-digit', month: 'short', year: 'numeric' });
}

function esc(s) {
    var d = document.createElement('div'); d.textContent = s || ''; return d.innerHTML;
}

function checkAuth() {
    var saved = localStorage.getItem('circul_operator');
    if (saved) {
        try {
            state.operator = JSON.parse(saved);
            if (state.operator.role !== 'processor') {
                var route = ROLE_ROUTES[state.operator.role];

function showDashboard() {
    document.getElementById('loginOverlay').classList.add('hidden');

    if (op.company && op.company.toLowerCase().indexOf('repatrn') >= 0) title = 'rePATRN — Processor';

function handleLogin(e) {
    e.preventDefault();
    var phone = document.getElementById('loginPhone').value.trim();
    var pin = document.getElementById('loginPin').value.trim();
    if (!phone || !pin) return;
    var btn = document.getElementById('loginBtn');
    btn.disabled = true; btn.innerHTML = '<div class="spinner"></div>';
    document.getElementById('loginError').classList.remove('visible');
    fetch('/api/operators/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ phone: phone, pin: pin }) })
    .then(function(r) { return r.json(); })
    .then(function(data) {
        if (data.success) {
            state.operator = data.operator;
            localStorage.setItem('circul_operator', JSON.stringify(data.operator));
            if (data.operator.role !== 'processor') {
                var route = ROLE_ROUTES[data.operator.role];
                if (route) { window.location.href = route; return; }
            }
            showDashboard();
        } else {
            var el = document.getElementById('loginError');
            el.textContent = data.message || 'Invalid phone or PIN';
            el.classList.add('visible');
        }
    })
    .catch(function() {
        var el = document.getElementById('loginError'); el.textContent = 'Connection error.'; el.classList.add('visible');
    })
    .finally(function() { btn.disabled = false; btn.textContent = 'Log In'; });
}

function handleLogout() {

function loadAll() {

function loadTxns() {
    fetch('/api/transactions?operator_id=' + state.operator.id + '&limit=30')
        .then(function(r) { return r.json(); })
        .then(function(data) {
            var tbody = document.getElementById('supplyBody');

                var statusClass = t.payment_status === 'paid' ? 'badge-green' : t.payment_status === 'pending' ? 'badge-amber' : 'badge-gray';

function loadAggregators() {
    fetch('/api/transactions?operator_id=' + state.operator.id + '&limit=100')
        .then(function(r) { return r.json(); })
        .then(function(data) {
            var el = document.getElementById('aggregatorRatingsList');
            if (!data.success || !data.transactions.length) { el.innerHTML = '<div class="empty">No transactions yet</div>'; return; }
            // Unique aggregators from transactions
            var seen = {};
            data.transactions.forEach(function(t) {
                if (t.operator_id && t.operator_name && !seen[t.operator_id]) seen[t.operator_id] = t.operator_name;
            });
            var aggs = Object.keys(seen);
            if (!aggs.length) { el.innerHTML = '<div class="empty">No aggregators to rate yet</div>'; return; }
            el.innerHTML = '<table style="width:100%;border-collapse:collapse;font-size:0.88rem;"><thead><tr><th style="padding:8px 0;border-bottom:1px solid var(--border);text-align:left;color:var(--slate-light);font-size:0.75rem;text-transform:uppercase">Aggregator</th><th style="padding:8px 0;border-bottom:1px solid var(--border)"></th></tr></thead><tbody>' +
                aggs.map(function(id) {
                    return '<tr><td style="padding:10px 0">' + esc(seen[id]) + '</td><td style="padding:10px 0;text-align:right"><button onclick="rateAggregator(' + id + ',\'' + esc(seen[id]) + '\')" style="padding:6px 14px;background:var(--green-primary);color:white;border:none;border-radius:6px;font-family:\'Space Grotesk\',sans-serif;font-size:0.78rem;font-weight:600;cursor:pointer">Rate</button></td></tr>';

        });
}

function rateAggregator(aggId, name) {
    var r = prompt('Rate ' + name + ' (1-5 stars):');
    if (!r || isNaN(parseInt(r)) || parseInt(r) < 1 || parseInt(r) > 5) return;
    fetch('/api/ratings/operator', {
        method: 'POST', headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ rater_operator_id: state.operator.id, rated_operator_id: aggId, rating: parseInt(r), rating_direction: 'processor_rates_aggregator' })
    }).then(function(r) { return r.json(); }).then(function(d) { if (d.success) alert('Rating submitted!'); });
}

function renderPriceCards(prices) {
    var priceMap = {};
    (prices || []).forEach(function(p) { priceMap[p.material_type] = p; });
    document.getElementById('priceCards').innerHTML = MATERIALS.map(function(mat) {
        var p = priceMap[mat];
        var current = p ? '$' + parseFloat(p.price_per_kg_usd).toFixed(4) : 'Not set';
        return '<div class="price-card">' +
            '<div class="price-card-mat">' + mat + '</div>' +
            '<div class="price-card-current">' + current + '</div>' +
            '<div class="price-input-row">' +
            '<input type="number" class="price-input" id="price-usd-' + mat + '" placeholder="USD/kg" step="0.0001" min="0">' +
            '<input type="number" class="price-input" id="price-rate-' + mat + '" placeholder="Rate" step="0.01" min="0" style="width:70px">' +
            '<button class="price-post-btn" onclick="postPrice(\'' + mat + '\')">Post</button>' +
            '</div></div>';
    }).join('');
}

function postPrice(mat) {
    var usd = document.getElementById('price-usd-' + mat).value;
    var rate = document.getElementById('price-rate-' + mat).value;
    if (!usd || parseFloat(usd) <= 0) { alert('Enter a valid USD price'); return; }
    var ghs = (rate && parseFloat(rate) > 0) ? (parseFloat(usd) * parseFloat(rate)).toFixed(2) : null;
    fetch('/api/prices', {
        method: 'POST', headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ operator_id: state.operator.id, material_type: mat, price_per_kg_usd: parseFloat(usd), price_per_kg_ghs: ghs ? parseFloat(ghs) : null, usd_to_ghs_rate: rate ? parseFloat(rate) : null })
    }).then(function(r) { return r.json(); }).then(function(data) { if (data.success) loadAll(); else alert(data.message); });
}

}

function checkAuth() {
    var saved = localStorage.getItem('circul_operator');
    if (saved) {
        try {
            state.operator = JSON.parse(saved);
            if (state.operator.role !== 'converter') {
                var route = ROLE_ROUTES[state.operator.role];
                if (route) { window.location.href = route; return; }
            }
            showDashboard();
            return;
        } catch (e) { localStorage.removeItem('circul_operator'); }
    }
    document.getElementById('loginOverlay').classList.remove('hidden');

function handleLogin(e) {
    e.preventDefault();
    var phone = document.getElementById('loginPhone').value.trim();
    var pin = document.getElementById('loginPin').value.trim();
    if (!phone || !pin) return;

    btn.disabled = true; btn.innerHTML = '<div class="spinner"></div>';

function rateProcessor(procId, name) {
    var r = prompt('Rate ' + name + ' (1-5):');
    if (!r || isNaN(parseInt(r)) || parseInt(r) < 1 || parseInt(r) > 5) return;

        method: 'POST', headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ rater_operator_id: state.operator.id, rated_operator_id: procId, rating: parseInt(r), rating_direction: 'converter_rates_processor' })

}

    }).join('');
}

function postPrice(mat) {
    var usd = document.getElementById('price-usd-' + mat).value;
    var rate = document.getElementById('price-rate-' + mat).value;
    if (!usd || parseFloat(usd) <= 0) { alert('Enter a valid USD price'); return; }
    var ghs = (rate && parseFloat(rate) > 0) ? (parseFloat(usd) * parseFloat(rate)).toFixed(2) : null;
    fetch('/api/prices', {
        method: 'POST', headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ operator_id: state.operator.id, material_type: mat, price_per_kg_usd: parseFloat(usd), price_per_kg_ghs: ghs ? parseFloat(ghs) : null, usd_to_ghs_rate: rate ? parseFloat(rate) : null })
    }).then(function(r) { return r.json(); }).then(function(data) { if (data.success) loadAll(); else alert(data.message); });
}

var MAT_COLORS = { PET: '#16a34a', HDPE: '#0891b2', LDPE: '#7c3aed', PP: '#f59e0b' };

function getParam(name) {
    var url = new URLSearchParams(window.location.search);
    return url.get(name);
}

function init() {
    var collectorId = getParam('id');
    var operatorId = getParam('operator_id');

    if (collectorId) {
        loadPassport(collectorId);
    } else if (operatorId) {
        // Try to find collector linked to this operator by looking up their transactions
        fetchCollectorForOperator(operatorId);
    } else {
        // Try from localStorage
        var saved = localStorage.getItem('circul_collector');
        if (saved) {
            try {
                var c = JSON.parse(saved);
                if (c.id) { loadPassport(c.id); return; }
            } catch (e) {}
        }
        showError('No collector ID provided. Add ?id=COLLECTOR_ID to the URL.');
    }
}

function fetchCollectorForOperator(opId) {
    // Look up transactions for this operator and find first collector
    fetch('/api/transactions?operator_id=' + opId + '&limit=1')
        .then(function(r) { return r.json(); })
        .then(function(data) {
            if (data.success && data.transactions && data.transactions.length) {
                loadPassport(data.transactions[0].collector_id);
            } else {
                showError('No transaction history found for this account. Log your first collection to build your passport.');
            }
        })
        .catch(function() { showError('Unable to load passport data.'); });
}

function loadPassport(id) {
    fetch('/api/collectors/' + id + '/passport')
        .then(function(r) { return r.json(); })
        .then(function(data) {
            if (!data.success) { showError(data.message || 'Passport not found'); return; }
            passportData = data.passport;
            renderPassport(data.passport);
        })
        .catch(function() { showError('Unable to load collector passport. Please try again.'); });
}
